Teksti u kopjua.');
  }catch(e){
    const ta = document.createElement('textarea');
    ta.value = protocolToText(p); document.body.appendChild(ta); ta.select(); document.execCommand('copy'); ta.remove();
    showProtocolToast('Teksti u kopjua.');
  }
}

function printProtocol(p=protocolFormData()){
  const body = protocolToText(p).split('\n').map(line => '<div>' + (line ? escapeHtml(line) : '&nbsp;') + '</div>').join('');
  const w = window.open('', '_blank', 'width=850,height=700');
  if(!w){ showProtocolToast('Shfletuesi e bllokoi dritaren e printimit.'); return; }
  w.document.write(`<!doctype html><html><head><meta charset="utf-8"><title>${escapeHtml(p.name || 'Recetë')}</title><style>body{font-family:Arial,sans-serif;max-width:850px;margin:35px auto;padding:0 24px;color:#17252a;line-height:1.5}div:first-child{font-size:24px;font-weight:700;margin-bottom:10px}div{white-space:pre-wrap} @media print{body{margin:0}}</style></head><body>${body}<script>window.onload=()=>window.print()<\/script></body></html>`);
  w.document.close();
}

function showProtocolToast(message){
  const toast = document.getElementById('protocolToast');
  if(!toast) return;
  toast.textContent = message;
  toast.classList.add('show');
  clearTimeout(showProtocolToast.t);
  showProtocolToast.t = setTimeout(() => toast.classList.remove('show'), 2400);
}
