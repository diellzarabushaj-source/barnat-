const PROTOCOL_STORAGE_KEY = 'regjistriBarnave_protokollet_v1';
let protocolMemoryStore = [];
const selectedDrugKeys = new Set();
let editingProtocolId = null;
let draftItems = [];

function drugKey(r){
  return String(r['PDID'] || '') + '|' + String(r['Emri tregtar'] || '') + '|' + String(r['Fortësia'] || '');
}

function getSavedProtocols(){
  try{
    const data = JSON.parse(localStorage.getItem(PROTOCOL_STORAGE_KEY) || '[]');
    if(Array.isArray(data)) protocolMemoryStore = data;
    return Array.isArray(data) ? data : protocolMemoryStore;
  }catch(e){ return protocolMemoryStore; }
}

function setSavedProtocols(protocols){
  protocolMemoryStore = protocols.map(p => ({...p, items:(p.items || []).map(i => ({...i}))}));
  try{ localStorage.setItem(PROTOCOL_STORAGE_KEY, JSON.stringify(protocols)); }catch(e){}
  updateProtocolCounters();
}

function updateProtocolCounters(){
  const selected = selectedDrugKeys.size;
  const saved = getSavedProtocols().length;
  const sc = document.getElementById('selectedCount');
  const pc = document.getElementById('savedProtocolCount');
  const tc = document.getElementById('savedTabCount');
  if(sc) sc.textContent = selected;
  if(pc) pc.textContent = saved;
  if(tc) tc.textContent = '(' + saved + ')';
}


function visibleColumns(){
  return COLUMNS.filter(c => c.visible);
}

function buildHeader(){
  const row = document.getElementById('headerRow');
  row.innerHTML = '';

  const selectTh = document.createElement('th');
  selectTh.className = 'select-col';
  selectTh.title = 'Zgjidhi të gjitha barnat në këtë faqe';
  const selectAll = document.createElement('input');
  selectAll.type = 'checkbox';
  selectAll.setAttribute('aria-label', 'Zgjidhi të gjitha barnat në këtë faqe');
  selectAll.addEventListener('click', e => e.stopPropagation());
  selectAll.addEventListener('change', () => toggleCurrentPageSelection(selectAll.checked));
  selectTh.appendChild(selectAll);
  row.appendChild(selectTh);

  visibleColumns().forEach(col => {
    const th = document.createElement('th');
    th.innerHTML = col.label + '<span class="arrow">' + (state.sortKey===col.key ? (state.sortDir===1?'▲':'▼') : '↕') + '</span>';
    if(state.sortKey===col.key) th.classList.add('sorted');
    th.addEventListener('click', () => {
      if(state.sortKey === col.key){ state.sortDir *= -1; }
      else { state.sortKey = col.key; state.sortDir = 1; }
      state.page = 1;
      render();
    });
    row.appendChild(th);
  });
}
