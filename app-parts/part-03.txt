const FORM_TO_CATEGORY = {};
CATEGORY_ORDER.forEach(cat => {
  FORM_CATEGORIES[cat].forEach(f => { FORM_TO_CATEGORY[f] = cat; });
});

function categoryOf(forma){
  return FORM_TO_CATEGORY[String(forma ?? '').trim()] || null;
}

function fieldMatchesTerm(fieldLower, term){
  if(fieldLower.includes(term)) return true;
  const aliasTargets = FORM_ALIASES[term];
  if(aliasTargets) return aliasTargets.some(t => fieldLower.includes(t));
  return false;
}

function rowMatchesSearch(r, terms){
  return terms.every(term => SEARCH_FIELDS.some(f => fieldMatchesTerm(String(r[f] ?? '').toLowerCase(), term)));
}

function getFiltered(){
  let rows = RAW;
  const q = state.search.trim().toLowerCase();
  if(q){
    const terms = q.split(/\s+/).filter(Boolean);
    rows = rows.filter(r => rowMatchesSearch(r, terms));
  }
  if(state.status){
    rows = rows.filter(r => String(r['Statusi'] ?? '').trim() === state.status);
  }
  if(state.formType === 'form'){
    rows = rows.filter(r => String(r['Forma farmaceutike'] ?? '').trim() === state.formValue);
  } else if(state.formType === 'category'){
    rows = rows.filter(r => categoryOf(r['Forma farmaceutike']) === state.formValue);
  }
  return rows;
}

function sortRows(rows){
  const col = COLUMNS.find(c => c.key === state.sortKey);
  const dir = state.sortDir;
  return [...rows].sort((a,b) => {
    let av = a[state.sortKey], bv = b[state.sortKey];
    if(col.type === 'num'){
      av = Number(av) || 0; bv = Number(bv) || 0;
      return (av - bv) * dir;
    }
    av = String(av ?? '').toLowerCase(); bv = String(bv ?? '').toLowerCase();
    return av.localeCompare(bv, 'sq') * dir;
  });
}

function escapeHtml(s){
  return String(s ?? '').replace(/[&<>"']/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]));
}

function qualityRowClass(row){
  const status = String(row.__qualityStatus || '').trim();
  return status && status !== 'verified' ? 'quality-' + status : '';
}

function qualityBadge(row){
  const status = String(row.__qualityStatus || '').trim();
  if(status === 'corrected') return '<span class="data-quality-badge corrected" title="' + escapeHtml(row.__qualityMessage || '') + '">✓ Korrigjuar</span>';
  if(status === 'blocked') return '<span class="data-quality-badge blocked" title="' + escapeHtml(row.__qualityMessage || '') + '">! Verifiko</span>';
  if(status === 'warning') return '<span class="data-quality-badge warning" title="' + escapeHtml(row.__qualityMessage || '') + '">Kontrollo</span>';
  return '';
}
